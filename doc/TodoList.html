def add(*new_todos)
  if new_todos.all? { |todo| todo.instance_of? Todo }
    @todos += new_todos
  else
    raise TypeError, "Can only pass Todo objects."
  end
end

def find_by_title(title_searched)
  title_found = nil

  each do |todo|
    if todo.title == title_searched
      title_found = todo
      break
    end
  end

  title_found
end
